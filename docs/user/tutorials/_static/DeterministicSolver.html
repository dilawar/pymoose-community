import moose
import pylab
import numpy
import sys
%matplotlib inline

def main():
        """ This example illustrates setting up of Deterministic solver, loading, running, and saving a kinetic model 
	defined in kkit format. """
	solver = "gsl"  # Setting up Deterministic solver
	mfile = '../genesis/Kholodenko.g'
	runtime = 6000.0
	modelId = moose.loadModel( mfile, 'model', solver )

	moose.reinit()
	moose.start( runtime ) 

	# Display all plots.
	for x in moose.wildcardFind( '/model/#graphs/conc#/#' ):
            t = numpy.arange( 0, x.vector.size, 1 ) * x.dt
            pylab.plot( t, x.vector, label=x.name )
        pylab.legend()
        pylab.show()

	quit()

# Run the 'main' if this script is executed standalone.
if __name__ == '__main__':
	main()
